51      private void etsiTiedostot() {
52          try {
53              File kansio = new File(tiedostoPolku + "/StockData");
54              tiedostot.clear();
55              for (File f : kansio.listFiles()) {
56                  if (!tiedostot.contains(f.getName())) {
57                      this.tiedostot.add(f.getName());
58                  }
59              }
60          } catch (Exception e) {
61  
62          }
63      }

259     private void HaeActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_HaeActionPerformed
260         String url = Url.getText();
261         String tiedostonNimi = luotavanTiedostonNimi.getText();
262 
263         boolean onnistui = true;
264         try {
265             dh.makeFileFromUrl(tiedostonNimi, url, dataHakuAlku.getText(), datanHakuLoppu.getText());
266         } catch (Exception ex) {
267             jTextArea1.setText("Jotain meni pieleen, tiedostoa ei luotu");
268             onnistui = false;
269         }
270         if (onnistui) {
271             jTextArea1.setText("Tiedosto luotiin!");
272         }
273 
274         etsiTiedostot();
275     }//GEN-LAST:event_HaeActionPerformed
